$qb = ActiveRecordsManager::getEntityManager(self::getNamespace())->createQueryBuilder();
$query = $qb->select($qb->expr()->count('op.id'))
	->from('\NS\AR\ABCP\OrderPosition', 'op')
	->where('op.Order = :order')
	->setMaxResults(1)
	->setParameters(array('order' => $order));
return $query->getQuery()->getSingleScalarResult();
