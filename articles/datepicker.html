// установка локализации
$.datepicker.setDefaults($.datepicker.regional['ru']);

$('.paymentDateTimeField').datepicker({
	showOn: 'both',
	buttonImage: '//astatic.abcp.ru/common.images/date_selector.gif',
	buttonImageOnly: true,
	changeMonth: true,
	changeYear: true,
	showOtherMonths: true,
	selectOtherMonths: true,
	dateFormat: 'dd.mm.yy',
	showAnim: 'slideDown'
});
Для выбора времени
