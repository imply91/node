В обеих функциях должен быть return 
$franchisee = \Franchisee::getCurrent();
$franchiseeDomain = (!empty($franchisee->domainAliases)) ?
	function() use ($franchisee) {
		$domainAliases = array_map('trim', explode(',', $franchisee->domainAliases));
		return !empty($domainAliases[0]) ? $domainAliases[0] : Registry::getSite()->officialSite;
	} :
	function() {
		return Registry::getSite()->officialSite;
	};
$templateValues['CONTACTS_SITE'] = $franchiseeDomain();
